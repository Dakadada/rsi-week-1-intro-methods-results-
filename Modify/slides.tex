\begin{frame}{Methods}
    \begin{block}{Graph Schema}
        \begin{itemize}
            \item Link peptide \& exp. cond. to observed phase
            \item Multi-agent extraction:
            \begin{itemize} 
                \item Section, entity, relation, graph, verifier, table, and classifier agents.
            \end{itemize}
        \end{itemize}
    \end{block}

    \pause

    \begin{center}
